    \paragraph*{9.}
        \textit{
            Funkiton g(x) määritellään asettamalla
        }
    \begin{equation*}
        g(x) =
        \begin{cases}
            x^2 -2, x <0\\
            3x + b, x\geq 0
        \end{cases}.
    \end{equation*}
    \textit{
        Millä parametrin b arvolla funktio on jatkuva pisteessä x = 0? piirrä tätä arvoa vastaava funkiton kuvaaja.
    }
    \\
    \begin{equation*}
        \begin{split}
            \lim_{x \rightarrow 0-} &=~x^2 -2\\
            &=~0^2-2\\
            &=~-2\\
            &=~b = -2\\
            \\
            \lim_{x \rightarrow 0+} &=~3x + b \\
            &=~3(0)+ (-2)\\
            &=~ -2 \\\\
            g(x) &= 
        \begin{cases}
            x^2 -2, x <0\\
            \cancel{3x + b, x\geq 0}\\
        \end{cases}
        \end{split}
    \end{equation*}
    \\
